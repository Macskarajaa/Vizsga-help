ArrayList<Bookings> bookings = new ArrayList<>();

    @FXML
    public VBox container;
    @FXML
    public void openFile() throws FileNotFoundException {
        Stage currentStage = (Stage) container.getScene().getWindow();

        FileChooser fileChooser = new FileChooser();
        fileChooser.setTitle("Fájl kiválasztása...");
        fileChooser.getExtensionFilters().addAll(new FileChooser.ExtensionFilter("CSV fájl", "*.csv"));

        File selectedFile = fileChooser.showOpenDialog(currentStage);

        Scanner scanner = new Scanner(selectedFile);

        scanner.nextLine();


        while(scanner.hasNextLine()){
            String line = scanner.nextLine();

            String[] parts = line.split(",");

            String customer_email = parts[0];
            String first_name = parts[1];
            String last_name = parts[2];
            String phone = parts[3];
            String birthday = parts[4];
            String customerImage = parts[5];

            int hotelId = Integer.parseInt(parts[6]);
            String hotelCity = parts[7];
            String hotelName = parts[8];
            String hotelDescription = parts[9];
            int hotelStars = Integer.parseInt(parts[10]);
            String hotelOwnerEmail = parts[11];
            String hotelImage = parts[12];

            String startDate = parts[13];
            String endDate = parts[14];

            int adults = Integer.parseInt(parts[15]);
            int children = Integer.parseInt(parts[16]);
            int boylovers = Integer.parseInt(parts[17]);

            bookings.add(new Bookings(customer_email, first_name, last_name, phone, birthday, customerImage, hotelId, hotelCity, hotelName, hotelDescription, hotelStars, hotelOwnerEmail, hotelImage, startDate, endDate, adults, children, boylovers));
  }

  @FXML protected void handleExit () {
        Platform.exit();
    }

    @FXML protected void handleAbout () {
        Alert alert = new Alert(Alert.AlertType.INFORMATION);
        alert.setTitle("Névjegy");
        alert.setHeaderText("");
        alert.setContentText("Vonatok v1.0.0\n(C) Kandó");
        alert.showAndWait();
    }
